# Popover

This is an overlay component that floats around its disclosure and is used to display additional information when triggered.

## Installation

<CodeGroup>

```bash {{ title: 'npm' }}
npm install @mojaui/popover
```

```bash {{ title: 'yarn' }}
yarn add @mojaui/popover
```

```bash {{ title: 'pnpm' }}
pnpm install @mojaui/popover
```

```bash {{ title: 'bun' }}
bun add @mojaui/popover
```

</CodeGroup>

## Usage

The Popover component can be used as follows:

```js
import { Popover } from '@mojaui/popover'

export function Index() {
  return (
    <Popover.Root>
      <Popover.Trigger></Popover.Trigger>
      <Popover.Content>
        <Popover.Arrow></Popover.Arrow>
        <Popover.Header>
          <Popover.Title></Popover.Title>
        </Popover.Header>
        <Popover.Body></Popover.Body>
        <Popover.Footer></Popover.Footer>
      </Popover.Content>
    </Popover.Root>
  )
}
```

## Controlled Popover

To create a controlled Popover component, you can use the `open` prop on the Root component.

You can monitor changes in the opened and closed states of the component using the `onOpenChange` callback prop.

```js
import { useState } from 'react'
import { Popover } from '@mojaui/popover'

export function Index() {
  const [open, setOpen] = useState(false)

  return (
    <>
      <button onClick={() => setOpen(!open)}>Click Me</button>
      <Popover.Root open={open} onOpenChange={(open) => console.log(open)}>
        <Popover.Content>
          <Popover.Arrow></Popover.Arrow>
          <Popover.Header>
            <Popover.Title></Popover.Title>
          </Popover.Header>
          <Popover.Body></Popover.Body>
          <Popover.Footer></Popover.Footer>
        </Popover.Content>
      </Popover.Root>
    </>
  )
}
```

## Placements

To change the placement of the popover, you can use the `position` prop.

```js
import { useState } from 'react'
import { Popover } from '@mojaui/popover'

export function Index() {
  return (
    <Popover.Root
      position={{
        placement: 'top',
        offset: 20,
      }}
    >
      <Popover.Content>
        <Popover.Arrow></Popover.Arrow>
        <Popover.Header>
          <Popover.Title></Popover.Title>
        </Popover.Header>
        <Popover.Body></Popover.Body>
        <Popover.Footer></Popover.Footer>
      </Popover.Content>
    </Popover.Root>
  )
}
```

## API Reference

### Root

<Properties>
  <Property name="closeOnEsc" type="string">
    Whether to close the popover when the escape key is pressed
  </Property>
  <Property name="closeOnOutsideClick" type="boolean">
    Whether to close the popover when clicked outside
  </Property>
  <Property name="defaultOpen" type="boolean">
    The initial open state of the popover when it is first rendered
  </Property>
  <Property name="open" type="boolean">
    Whether the popover is open
  </Property>
  <Property name="onOpenChange" type="(details: OpenChangeDetails) => void">
    Function to call when the popover opens or closes
  </Property>
  <Property name="position" type="PositionOptions">
    Options used to dynamically position the popover
  </Property>
  <Property name="[data-state]"> "open" | "closed" </Property>
</Properties>

### Content

<Properties>
  <Property name="[data-state]"> "open" | "closed" </Property>
  <Property name="[data-placement]"> Placement of the content </Property>
</Properties>

## Accessibility

### Keyboard Support

<Properties>
  <Property name="Space | Enter">Opens/Closes the popover</Property>
  <Property name="Tab">
    Moves focus to the next focusable element within the content. If there are
    no focusable elements, focus is moved to the next focusable element after
    the trigger
  </Property>
  <Property name="Shift + Tab">
    Moves focus to the previous focusable element within the content. If there
    are no focusable elements, focus is moved to the next focusable element
    after the trigger
  </Property>
  <Property name="Esc">
    Closes the popover and moves focus to the trigger
  </Property>
</Properties>
